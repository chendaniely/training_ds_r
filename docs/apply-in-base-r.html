7.5.4 mapply

v1 <- c(1, 2, 3, 4)
v2 <- c(100, 200, 300, 400)
my_mean <- function(x, y){
    return((x + y) / 2)
}
# sapply(v1, v2, my_mean)
mapply(my_mean, v1, v2)
## [1]  50.5 101.0 151.5 202.0
# this is the same as purrr::map2
